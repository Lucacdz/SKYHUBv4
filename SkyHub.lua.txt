local success, error = pcall(function()
    loadstring(game:HttpGet('https://raw.githubusercontent.com/Lucacdz/SKYHUBv4/main/SkyHub.lua.txt'))()
end)

if not success then
    warn("Lỗi khi chạy loadstring: " .. tostring(error))
    
    -- Hiển thị thông báo lỗi
    local ErrorGui = Instance.new("ScreenGui")
    ErrorGui.Name = "ErrorGUI"
    ErrorGui.Parent = game:GetService("CoreGui")
    
    local ErrorFrame = Instance.new("Frame")
    ErrorFrame.Size = UDim2.new(0, 300, 0, 150)
    ErrorFrame.Position = UDim2.new(0.5, -150, 0.5, -75)
    ErrorFrame.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    ErrorFrame.BorderSizePixel = 0
    ErrorFrame.ZIndex = 1000
    ErrorFrame.Parent = ErrorGui
    
    local ErrorText = Instance.new("TextLabel")
    ErrorText.Text = "❌ Lỗi khi tải SkyHub\n\n" .. tostring(error)
    ErrorText.Size = UDim2.new(1, -20, 1, -20)
    ErrorText.Position = UDim2.new(0, 10, 0, 10)
    ErrorText.BackgroundTransparency = 1
    ErrorText.TextColor3 = Color3.fromRGB(255, 100, 100)
    ErrorText.Font = Enum.Font.GothamBold
    ErrorText.TextSize = 14
    ErrorText.TextWrapped = true
    ErrorText.ZIndex = 1001
    ErrorText.Parent = ErrorFrame
    
    -- Tự động ẩn sau 10 giây
    wait(10)
    ErrorGui:Destroy()
else
    print("✅ SkyHub đã được tải thành công!")
end
